1. Application URL:
   {{- range .Values.ingress.hosts }}
   https://{{ .host }}

1. Get the API URL by running:
   {{- if contains "NodePort" .Values.api.service.type }}
   export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "neurolake.fullname" . }}-api)
   export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
   echo http://$NODE_IP:$NODE_PORT
   {{- else if contains "LoadBalancer" .Values.api.service.type }}
   export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "neurolake.fullname" . }}-api -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
   echo http://$SERVICE_IP:{{ .Values.api.service.port }}
   {{- else if contains "ClusterIP" .Values.api.service.type }}
   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "neurolake.fullname" . }}-api {{ .Values.api.service.port }}:{{ .Values.api.service.port }}
   echo "API: http://127.0.0.1:{{ .Values.api.service.port }}"
   {{- end }}

2. Get the Frontend URL by running:
   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "neurolake.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }}
   echo "Frontend: http://127.0.0.1:{{ .Values.frontend.service.port }}"
{{- end }}
